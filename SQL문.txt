CREATE TABLE UNKNOWN_BOARD(
	BOARD_NO NUMBER PRIMARY KEY,
	BOARD_TITLE NVARCHAR2(30) NOT NULL,	
	BOARD_CONTENT NVARCHAR2(200) NOT NULL,
	BOARD_QUESTION NUMBER NOT NULL,
	BOARD_PWD VARCHAR2(30) NOT NULL,
	BOARD_HINT NVARCHAR2(30),
	FILE_URL VARCHAR2(1000),
	CREATE_DATE TIMESTAMP DEFAULT SYSDATE,
	MODIFIED CHAR(1) DEFAULT 'N' CHECK (MODIFIED IN ('Y','N'))
);


CREATE TABLE CATEGORY(

CREATE TABLE ADMIN(
	ADMIN_NO NUMBER PRIMARY KEY,
	ADMIN_ID VARCHAR2(20) NOT NULL,
	ADMIN_PWD VARCHAR2(200) NOT NULL,
	ADMIN_NAME NVARCHAR2(20) NOT NULL,
	ROLE VARCHAR(20) DEFAULT 'ROLE_ADMIN',
	CREATE_DATE TIMESTAMP DEFAULT SYSDATE,
	STATUS CHAR(1) DEFAULT 'Y' CHECK(STATUS IN('Y', 'N'))
);

CREATE TABLE ADMIN_TOKEN(
	ADMIN_NO NUMBER NOT NULL REFERENCES ADMIN,
	TOKEN VARCHAR2(300) NOT NULL,
	EXPIRATION NUMBER NOT NULL
);
